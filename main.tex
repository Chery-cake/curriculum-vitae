% ---------------------------------------------------------------------------
% Conditional section inclusion helper
% ---------------------------------------------------------------------------
% \cvsection{Title}{sections/filename.tex}
% Reads the file, strips comment-only lines, and includes the section only
% if there is real (non-comment) content. Empty or comment-only files produce
% no output and no section header.
% ---------------------------------------------------------------------------
\newread\testread
\newcommand{\cvsection}[2]{%
  % \endlinechar=-1 suppresses end-of-line chars; \catcode`\%=14 treats % as
  % a comment character so comment-only lines are stripped during the read.
  \CatchFileDef{\filecontent}{#2}{\endlinechar=-1 \catcode`\%=14 }%
  \ifx\filecontent\empty\else
    \section*{#1}%
    \input{#2}%
  \fi
}
